9/*
10 * https://docs.fileformat.com/audio/wav/
11 * Positions Sample Value Description
121 - 4 “RIFF” Marks the file as a riff file. Characters are each 1 byte long.
135 - 8 File size (integer) Size of the overall file - 8 bytes, in bytes (32-bit integer). Typically, you’d fill this in after creation.
149 -12 “WAVE” File Type Header. For our purposes, it always equals “WAVE”.
1513-16 “fmt " Format chunk marker. Includes trailing null
1617-20 16 Length of format data as listed above
1721-22 1 Type of format (1 is PCM) - 2 byte integer
1823-24 2 Number of Channels - 2 byte integer
1925-28 44100 Sample Rate - 32 byte integer. Common values are 44100 (CD), 48000 (DAT). Sample Rate = Number of Samples per second, or Hertz.
2029-32 176400 (Sample Rate * BitsPerSample * Channels) / 8.
2133-34 4 (BitsPerSample * Channels) / 8.1 - 8 bit mono2 - 8 bit stereo/16 bit mono4 - 16 bit stereo
2235-36 16 Bits per sample
2337-40 “data” “data” chunk header. Marks the beginning of the data section.
2441-44 File size (data) Size of the data section.
25 */
26typedef struct wav_header {
27 // RIFF Header
28 char riff_header[4]; // Contains "RIFF"
29 int wav_size; // Size of the wav portion of the file, which follows the first 8 bytes. File size - 8
30 char wave_header[4]; // Contains "WAVE"
31
32 // Format Header
33 char fmt_header[4]; // Contains "fmt " (includes trailing space)
34 int fmt_chunk_size; // Should be 16 for PCM
35 short audio_format; // Should be 1 for PCM. 3 for IEEE Float
36 short num_channels; // Mono = 1, Stereo = 2
37 int sample_rate;
38 int byte_rate; // Number of bytes per second. sample_rate * num_channels * Bytes Per Sample
39 short sample_alignment; // num_channels * Bytes Per Sample
40 short bit_depth; // Number of bits per sample
41
42 // Data
43 char data_header[4]; // Contains "data"
44 int data_bytes; // Number of bytes in data. Number of samples * num_channels * sample byte size
45 // char bytes[]; // Remainder of wave file is bytes
46} wav_header;
